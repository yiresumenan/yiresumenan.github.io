2021-11-15

动态 URL 传参

hello_app/urls.py

from django.urls import path

from . import views

urlpatterns = [
path(‘articles/int:year/‘, views.year_archive),
path(‘articles/int:year/int:month/‘, views.month_archive),
path(‘articles/int:year/int:month/slug:title/‘, views.article_title),
]
